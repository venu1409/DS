Graph Operations

Explore algorithms and operations performed on Graphs, including representation, traversal, and shortest path calculations.

Graph Representation
(Adjacency Matrix)
Algorithm
1. Initialize a 2D array matrix of size V x V with 0. 2. For each edge connecting vertex u to v, set matrix[u][v] = 1. 3. For undirected graphs, also set matrix[v][u] = 1.
Code 
#include <stdio.h>
#define V 5

int adjMatrix[V][V];

void initMatrix() {
    for(int i = 0; i < V; i++)
        for(int j = 0; j < V; j++)
            adjMatrix[i][j] = 0;
}

void addEdge(int u, int v) {
    adjMatrix[u][v] = 1;
    adjMatrix[v][u] = 1; // For undirected graph
}
 O(V²)
Breadth First Search
(BFS)
Algorithm
1. Create a queue and a visited array. 2. Mark the starting node as visited and enqueue it. 3. While the queue is not empty: - Dequeue a node and visit/print it. - Enqueue all its unvisited adjacent nodes and mark them visited.
Code 
#include <stdio.h>

int queue[100], front = -1, rear = -1;
int visited[5];

void enqueue(int val) {
    if(front == -1) front = 0;
    queue[++rear] = val;
}

int dequeue() {
    int item = queue[front];
    if(front == rear) front = rear = -1;
    else front++;
    return item;
}

void bfs(int adjMatrix[5][5], int startNode, int n) {
    enqueue(startNode);
    visited[startNode] = 1;

    while(front != -1) {
        int curr = dequeue();
        printf("%d ", curr);

        for(int i = 0; i < n; i++) {
            if(adjMatrix[curr][i] == 1 && !visited[i]) {
                enqueue(i);
                visited[i] = 1;
            }
        }
    }
}
 O(V + E)
Depth First Search
(DFS)
Algorithm
1. Start from the target vertex. 2. Mark the vertex as visited and print it. 3. For each adjacent vertex of this node, recursively call DFS if it has not been visited yet.
Code 
#include <stdio.h>

int visited[5];

void dfs(int adjMatrix[5][5], int node, int n) {
    visited[node] = 1;
    printf("%d ", node);

    for(int i = 0; i < n; i++) {
        if(adjMatrix[node][i] == 1 && !visited[i]) {
            dfs(adjMatrix, i, n);
        }
    }
}
 O(V + E)
Dijkstra's Shortest Path
Algorithm
1. Create a dist array (initialized to infinity, source = 0) and a sptSet (Shortest Path Tree) boolean array. 2. Loop V - 1 times: - Select vertex u not in sptSet with minimum distance. - Include u in sptSet. - Update distance values of all adjacent vertices of u.
Code 
#include <stdio.h>
#include <limits.h>
#define V 5

int minDistance(int dist[], int sptSet[]) {
    int min = INT_MAX, min_index;
    for (int v = 0; v < V; v++)
        if (sptSet[v] == 0 && dist[v] <= min)
            min = dist[v], min_index = v;
    return min_index;
}

void dijkstra(int graph[V][V], int src) {
    int dist[V];
    int sptSet[V];

    for (int i = 0; i < V; i++)
        dist[i] = INT_MAX, sptSet[i] = 0;

    dist[src] = 0;

    for (int count = 0; count < V - 1; count++) {
        int u = minDistance(dist, sptSet);
        sptSet[u] = 1;

        for (int v = 0; v < V; v++)
            if (!sptSet[v] && graph[u][v] && dist[u] != INT_MAX
                && dist[u] + graph[u][v] < dist[v])
                dist[v] = dist[u] + graph[u][v];
    }
}
 O(V²)
