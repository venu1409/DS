Binary Search

Learn about the operations, implementation, and time complexity of Binary Search.

Algorithm Code Time Complexity
1. Start

2. Read n (number of elements)

3. Read Sorted array elements

4. Read element to search (key)

5. Set low = 0, high = n - 1

6. While low <= high :
mid = (high + low) / 2

    If arr[mid] == key
        print "Element Found"
        break

    Else if key < arr[mid]
        high = mid - 1

    Else
        low = mid + 1

7. If not found, print "Element not found"
8. Stop
 
Without Function:    
 
#include <stdio.h>

int main()
{
    int arr[100], n, key;
    int low, high, mid, found = 0;

    printf("Enter number of elements: ");
    scanf("%d", &n);

    printf("Enter sorted array elements:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    printf("Enter element to search: ");
    scanf("%d", &key);

    low = 0;
    high = n - 1;

    while(low <= high)
    {
        mid = (low + high) / 2;

        if(arr[mid] == key)
        {
            printf("Element Found at Position %d", mid + 1);
            found = 1;
            break;
        }
        else if(key < arr[mid])
        {
            high = mid - 1;
        }
        else
        {
            low = mid + 1;
        }
    }

    if(found == 0)
        printf("Element Not Found");

    return 0;
}

With Function :

#include <stdio.h>

int binarySearch(int arr[], int n, int key)
{
    int low = 0, high = n - 1;

    while(low <= high)
    {
        int mid = (low + high) / 2;

        if(arr[mid] == key)
            return mid;
        else if(key < arr[mid])
            high = mid - 1;
        else
            low = mid + 1;
    }

    return -1;
}

int main()
{
    int arr[] = {10, 20, 30, 40, 50, 60, 70};
    int n = 7;
    int key = 40;

    int result = binarySearch(arr, n, key);

    if(result == -1)
        printf("Element Not Found");
    else
        printf("Element Found at Position %d", result + 1);

    return 0;
}
O(log n)
