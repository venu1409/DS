Circularly Linked List Operations

Learn about the operations, implementation, and time complexity of Circularly Linked List Operations.

Operation Algorithm (Text / Code) Time Complexity
Node Structure 
struct Node
{
    int data;
    struct Node *next;
};

struct Node *head = NULL;
1. Create a node
2. Store data in node
3. Store address of next node in next
4. Last node points back to head instead of NULL
No Time Complexity
Traversal
Algorithm
1. Start from head
2. Print node
3. Move to next node
4. Stop when head is reached again
Code 
void display()
{
    if(head == NULL)
        return;

    struct Node *temp = head;

    do
    {
        printf("%d ", temp->data);
        temp = temp->next;
    }
    while(temp != head);
}
O(n)
Insertion (Beginning)
Algorithm
1. Create a new node
2. If list is empty:
Make node point to itself
Make it head
3. Otherwise:
Traverse to last node
Set new node's next as head
Set last node's next as new node
Update head
Code 
void insertBeginning(int data)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;

    if(head == NULL)
    {
        newNode->next = newNode;
        head = newNode;
        return;
    }

    struct Node *temp = head;

    while(temp->next != head)
        temp = temp->next;

    newNode->next = head;
    temp->next = newNode;
    head = newNode;
}
 O(n)

Algorithm
1. Create a new node
2. If list empty:
Make node point to itself
Make it head
3. Traverse to last node
4. Link last node to new node
5. Link new node to head

Insertion (At Specific Position)
Algorithm
1. Create a new node
2. Traverse to position-1
3. Insert node between nodes
4. Adjust links
Code 
void insertPosition(int data, int pos)
{
    struct Node *newNode =
        (struct Node*)malloc(sizeof(struct Node));

    newNode->data = data;

    struct Node *temp = head;

    for(int i=1;i < pos-1 ; i++)
        temp = temp->next;

    newNode->next = temp->next;
    temp->next = newNode;
}
 O(n)
Deletion (At Beginning)
Algorithm
1. If only one node:
Delete node
Set head = NULL
2. Traverse to last node
3. Move head to next node
4. Make last node point to new head
5. Delete old head
Code 
void deleteBeginning()
{
    if(head == NULL)
        return;

    if(head->next == head)
    {
        free(head);
        head = NULL;
        return;
    }

    struct Node *temp = head;
    struct Node *last = head;

    while(last->next != head)
        last = last->next;

    head = head->next;
    last->next = head;

    free(temp);
}
O(n)
Deletion (At End)
Algorithm
1. Traverse to last node
2. Move to previous node
3. Set prev node point to head
4. Delete last node
Code 
void deleteEnd()
{
    if(head == NULL)
        return;

    if(head->next == head)
    {
        free(head);
        head = NULL;
        return;
    }

    struct Node *temp = head;

    while(temp->next->next != head)
        temp = temp->next;

    free(temp->next);

    temp->next = head;
}
O(n)
Deletion (At Specific Position)
Algorithm
1. Traverse to position-1
2. Store node to delete
3. Skip that node
4. Free deleted node
Code 
void deletePosition(int pos)
{
    struct Node *temp = head;

    for(int i=1;i < pos-1;i++)
        temp = temp->next;

    struct Node *del = temp->next;

    temp->next = del->next;

    free(del);
}
O(n)
Searching a node
Algorithm
1. Start from head
2. Compare each node with key
3. If found, return position
4. Otherwise continue untill head is reached again
Code 
void search(int key)
{
    if(head == NULL)
        return;

    struct Node *temp = head;
    int pos = 1;

    do
    {
        if(temp->data == key)
        {
            printf("Found at Position %d", pos);
            return;
        }

        temp = temp->next;
        pos++;
    }
    while(temp != head);

    printf("Not Found");
}
O(n)
Updating a node
Algorithm
1. Traverse list
2. Find old value
3. Replace with new value
Code 
void update(int oldValue, int newValue)
{
    if(head == NULL)
        return;

    struct Node *temp = head;

    do
    {
        if(temp->data == oldValue)
        {
            temp->data = newValue;
            return;
        }

        temp = temp->next;
    }
    while(temp != head);
}
O(n)
Reversing
Algorithm
1. Initialise:
prev = NULL
current = head
2. Reverse links one by one
3. Continue untill current reaches head again
4. Adjust last node and head
Code 
void reverse()
{
    if(head == NULL || head->next == head)
        return;

    struct Node *prev = NULL;
    struct Node *current = head;
    struct Node *nextNode;

    do
    {
        nextNode = current->next;
        current->next = prev;

        prev = current;
        current = nextNode;
    }
    while(current != head);

    head->next = prev;
    head = prev;
}
O(n)
Count number of nodes
Algorithm
1. Initialize count = 0
2. Traverse once around the circle
3. Increment count for each node
4. return count
Code 
int countNodes()
{
    if(head == NULL)
        return 0;

    int count = 0;
    struct Node *temp = head;

    do
    {
        count++;
        temp = temp->next;
    }
    while(temp != head);

    return count;
}
O(n)
