Heap Sort

Learn about the operations, implementation, and time complexity of Heap Sort.

Algorithm Code Time Complexity
1. Start

2. Build a max heap from the array

3. Swap the root (largest element) with the last element

4. Reduce Heap Size by 1

5. Heapify the root again

6. Repeat steps 3 and 5 untill heap size becomes 1

7. Stop

 
Without Function:

#include <stdio.h>

void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}

void heapify(int arr[], int n, int i)
{
    int largest = i;
    int left = 2 * i + 1;
    int right = 2 * i + 2;

    if(left < n && arr[left] > arr[largest])
        largest = left;

    if(right < n && arr[right] > arr[largest])
        largest = right;

    if(largest != i)
    {
        swap(&arr[i], &arr[largest]);
        heapify(arr, n, largest);
    }
}

void heapSort(int arr[], int n)
{
    int i;

    // Build max heap
    for(i = n / 2 - 1; i >= 0; i--)
        heapify(arr, n, i);

    // One by one remove elements
    for(i = n - 1; i > 0; i--)
    {
        swap(&arr[0], &arr[i]);
        heapify(arr, i, 0);
    }
}

int main()
{
    int arr[] = {50, 20, 40, 10, 30};
    int n = sizeof(arr) / sizeof(arr[0]);

    heapSort(arr, n);

    printf("Sorted array:\n");
    for(int i = 0; i < n; i++)
        printf("%d ", arr[i]);

    return 0;
}
